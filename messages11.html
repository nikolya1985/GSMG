import sympy

def break_into_primes(s):
primes = []
start = 0

while start < len(s):
for end in range(start + 1, len(s) + 1):
num = int(s[start:end])
if sympy.isprime(num):
primes.append(num)
start = end
break
# If we're at the end of the string and haven't found a prime, we need to backtrack
if end == len(s):
start -= 1
if start < 0:
raise ValueError("Couldn't break string into primes.")
del primes[-1]
return primes
